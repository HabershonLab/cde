123  subroutine dfp_lnsrch(n,xold,fold,g,p ,x ,f ,stpmax,lcheck,calcf,cx,success)
124  implicit none
125  integer(4), intent(in) :: n
126  logical, intent(inout) :: lcheck, success
127  double precision, intent(in) :: stpmax,fold
128  double precision, intent(out) :: f
129  double precision, dimension(n), intent(in) :: g,xold
130  double precision, dimension(n), intent(out) :: x
131  double precision, dimension(n), intent(inout) :: p
132  integer(4) :: i,idx
133  double precision :: alf,tolx
134  double precision :: a,alam,alam2,alamin,b,disc,f2,fold2,rhs1,rhs2,slope,sum,temp,test,tmplam
135  procedure() :: calcf
136  type(cxs) :: cx
137 
138  alf = 1.0d-4
139  tolx = 1.0d-5
140  !tolx = 1.0d-7
141  lcheck = .false.
142  sum = dot_product(p,p)
143  sum = dsqrt(sum)
144  if(sum.gt.stpmax)then
145  print*, 'SUM EXCEEDED STPMAX'
146  p = p*stpmax/sum
147  endif
148  slope = dot_product(g,p)
149  test = 0.0d0
150  do i = 1,n
151  temp = abs(p(i))/max(abs(xold(i)),1.0d0)
152  if(temp.gt.test)test = temp
153  enddo
154  alamin = tolx/test
155  alam = 1.0d0
156  idx = 0
157  do
158  !write(6,*) cx%na ; write(6,*) 'GET '
159  !do i = 1, cx%na
160  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
161  !enddo
162  !cx%r = reshape(xold,(/3,cx%na/))
163  !write(6,*) cx%na ; write(6,*) 'XOLD : '
164  !do i = 1, cx%na
165  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
166  !enddo
167  !cx%r = reshape(p,(/3,cx%na/))
168  !write(6,*) cx%na ; write(6,*) 'P : '
169  !do i = 1, cx%na
170  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
171  !enddo
172  idx = idx + 1
173  x(:) = xold(:)+alam*p(:)
174  cx%r = reshape(x,(/3,cx%na/))
175  !print*, 'ALAM = ', alam, 'SUM = ', sum, stpmax
176  !write(6,*) cx%na ; write(6,*) 'AFTER ALAM '
177  !do i = 1, cx%na
178  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
179  !enddo
180  call calcf(cx,success)
181  !print*, 'SUC = ', success, cx%vcalc, alam
182  !if (.not. success) return
183  f = cx%vcalc
184  If(alam.lt.alamin)then
185  x(:) = xold(:)
186  lcheck = .true.
187  !write(*,*)'Convergence on dx: ',idx
188  !write(6,*) cx%na ; write(6,*) ' RETURN '
189  !do i = 1, cx%na
190  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
191  !enddo
192  return
193  else if(f .le. (fold+alf*alam*slope))then
194  !write(*,*)'Sufficient decrease ',idx
195  !write(6,*) cx%na ; write(6,*) ' RETURN '
196  !do i = 1, cx%na
197  ! write(6,'("'//cx%atomlabel(i)//'",3F8.4)') cx%r(1:3,i)
198  !enddo
199  return
200  else
201  if(abs(alam-1.0d0) .lt. 1.0d-10)then
202  tmplam = -1.0d0*slope/(2.0d0*(f-fold-slope))
203  else
204  rhs1 = f - fold - alam*slope
205  rhs2 = f2 - fold2 - alam2*slope
206  a = (rhs1/alam**2 - rhs2/alam2**2) / (alam-alam2)
207  b = (-1.0d0*alam2*rhs1/alam**2 + alam*rhs2/alam2**2) &
208  / (alam-alam2)
209  !print*, 'ab = ', a,b, slope
210  if(abs(a).lt.1.0d-10)then
211  tmplam = -1.0d0*slope / (2.0d0*b)
212  else
213  disc = b*b - 3.0d0*a*slope
214  if(disc.lt.0.0d0) then
215  !print*, 'DISK = ', disc
216  !write(*,*)'Problem in dfp_lnsrch: roundoff'
217  disc = abs(disc)*0.10d0
218  !stop
219  endif
220  tmplam = (-1.0d0*b + dsqrt(disc)) / (3.0d0*a)
221  endif
222  if(tmplam.gt.0.5d0*alam) tmplam=0.5d0*alam
223 
224  endif
225  endif
226  alam2 = alam
227  f2 = f
228  fold2 = fold
229  alam = max(tmplam,0.1d0*alam)
230  enddo
231  return
232  end subroutine dfp_lnsrch

370  subroutine dfp_stp2(dfp)
371  implicit none
372  integer :: i,j,k,l,m,n,o
373  type(vtdfp) :: dfp
374  double precision :: test, temp, den, fac, fae, sumxi, sumdg, fad
375  double precision :: hdg(dfp%n)
376  Character(len=10) :: ftypin
377  logical :: check
378 
379  !write(6,*) dfp%cx%na ; write(6,*) ' JUST INSIDE '
380  !do i = 1, dfp%cx%na
381  ! write(6,'("'//dfp%cx%atomlabel(i)//'",3F8.4)') dfp%cx%r(1:3,i)
382  !enddo
383  dfp%g = reshape(dfp%cx%dvdr,(/dfp%n/))
384  ! adds constraints...
385  do i = 1, dfp%n
386  if( dfp%cx%fixedatom( (i-1)/3 + 1 ) .or. dfp%cx%fixeddof(i) ) dfp%g(i) = 0.0d0
387  enddo
388  den = max(dfp%fret,1.0d0)
389  dfp%maxg = 0.0d0
390  do i = 1,dfp%n
391  temp=abs(dfp%g(i))*max(abs(dfp%p(i)),1.0d0)/den
392  if(temp.gt.dfp%maxg)dfp%maxg=temp
393  enddo
394  if(dfp%maxg.lt.dfp%gtol) dfp%lgtol = .true.
395  ! update hessin:
396  ! note that xi here is not the force vector but the difference in positions (from stp2 or init)
397  dfp%dg = dfp%g - dfp%dg
398  hdg = matmul(dfp%hessin,dfp%dg)
399  fac = dot_product(dfp%dg,dfp%xi)
400  fae = dot_product(dfp%dg,hdg)
401  sumdg = dot_product(dfp%dg,dfp%dg)
402  sumxi = dot_product(dfp%xi,dfp%xi)
403  if (mod(dfp%its,dfp%resetfreq) .eq. 0 ) then
404  dfp%hessin = 0.0d0
405  do i = 1, dfp%n ; dfp%hessin(i,i) = 1.0d0 ; enddo
406  dfp%dg = reshape(dfp%cx%dvdr,(/dfp%n/))
407  dfp%xi = -dfp%dg
408  endif
409  if(fac*fac .gt. dfp%eps*sumdg*sumxi)then
410  fac = 1.0d0/fac
411  fad = 1.0d0/fae
412  do i = 1,dfp%n ! this is the BFGS line:
413  dfp%dg(i) = fac*dfp%xi(i)-fad*hdg(i)
414  enddo
415  do i = 1,dfp%n
416  do j = 1,dfp%n
417  dfp%hessin(i,j) = dfp%hessin(i,j) + fac*dfp%xi(i)*dfp%xi(j) &
418  - fad*hdg(i)*hdg(j) + fae*dfp%dg(i)*dfp%dg(j)
419  enddo
420  enddo
421  endif
422  !print*, 'HESSIAN UPDATED'
423  !do i = 1, dfp%n
424  ! do j = 1, dfp%n
425  ! write(6,'(F8.1," ")',advance='no') dfp%hessin(i,j)
426  ! enddo
427  ! write(6,*)
428  !enddo
429  !print*,'NORM DG = ', norm2(dfp%dg)
430  !print*, 'LENG = ', norm2(dfp%g)
431  ! choose a step size in case cnt >=5
432 
433  dfp%gg = dfp%g
434  dfp%xi = -matmul(dfp%hessin,dfp%g)
435  if (norm2(dfp%g)/norm2(dfp%xi) .lt. 0.05d0) then
436  dfp%xi = 20.0d0*norm2(dfp%g) * dfp%xi/norm2(dfp%xi)
437  endif
438  return
439  end subroutine
