40  Subroutine optimisemolecules(cx,optall,success,nopti,midi)
41  implicit none
42  Type(cxs) :: cx
43  type(cxs), allocatable :: mcx(:)
44  integer :: i, j, k, l, j1, l1, m, n, nd, nb, naa, imove, ic, nactmol(2), icc(2), infos, itout, itmp, nopt
45  integer :: nrx, ingraph(cx%na,cx%na), m1
46  logical :: success, matchgraph, fixedatoms(cx%na), dontrotate, ldum, esscheck, backupreactive(NAMAX), optall
47  integer, allocatable :: ograph(:,:), rxindex(:), mid(:)
48  logical, allocatable :: bondchanges(:,:), atomchange(:)
49  real(8), allocatable :: or(:,:), D(:,:), U(:,:), E(:), s(:), VT(:,:), work(:)
50  real(8) :: cm1(1:3), Ma, cm2(1:3), unt(3,3)
51  character(10) :: ww
52  character(50) :: filen
53  integer, optional :: nopti
54  integer, optional :: midi(*)
55  ! for every molecule in mid(1:nopt)
56  ! save cartesian coordinates, centre of mass.
57  ! optimise molecule structure
58  ! reorient molecule so that its in the same orientation as the old way.
59  ! put molecules back in the soup and relax the other molecules away from the new ones
60  ! if the graph has changed, we have to recalculate the new molecular properties and perform accepatnce tests..
61  success = .true.
62  if (optall .or. .not. present(nopti) .or. .not. present(midi) ) then
63  nopt = cx%nmol
64  allocate(mid(cx%nmol))
65  do i = 1, nopt ; mid(i) = i ; enddo
66  write(logfile,'("Optimising all ",I5," Molecules")') nopt
67  else
68  nopt = nopti
69  allocate(mid(nopt))
70  mid(1:nopt) = midi(1:nopti)
71  write(logfile,'("Optimising ",I5," Specific Molecules")') nopt
72  endif
73  itmp = itmp + 1
74  ! copy original graph
75  matchgraph = .true.
76  ingraph(:,:) = cx%graph
77  unt = 0.0d0 ; unt(1,1) = 1.0d0 ; unt(2,2) = 1.0d0 ; unt(3,3) = -1.0d0
78  allocate(mcx(nopt))
79  itout = 0
80  Call gcminimisecollision(cx, ngdsrelax, gdsdtrelax, kradius )
81  write(logfile,"('* : Molecules Optimised - ',I2)") nopt
82  do m1 = 1, nopt ; m = mid(m1) ! for active molecules in fron and tail
83  Call createmolecularcx(cx,mcx(m1),m)
84  naa = mcx(m1)%na ; nd = naa*3
85  if (allocated(ograph)) deallocate(ograph,or,d,u,vt,e,work)
86  allocate(ograph(naa,naa),or(3,naa),d(3,3),u(3,3),vt(3,3),e(3),work(nd*100))
87  ograph = mcx(m1)%graph
88  or = mcx(m1)%r
89  cm1 = 0.0d0 ; cm2 = 0.0d0 ; ma = 0.0d0
90  do l = 1, naa
91  ma = ma + mcx(m1)%mass(l)
92  enddo
93  n = 0
94  ! minimise molecular geometries. some programs have not been implemented
95  ! with atom and dof constraints
96  write(logfile,'("* : Molecule Optimised - ",A10)') molecularformula(mcx(m1), 1)
97  if (nebrestrend) then
98  !write(logfile,'("* : Running CDE Constrained Optimization (Fire) - ")')
99  !Call FireMinimise(mcx(m1),n, gdsrestspring, nbstrength, nbrange, kradius, nebrestrend )
100  write(logfile,'("* : Running CDE Optimization (BFGS) - ")')
101  Call dfpminimise(mcx(m1),n,nebrestrend )
102  write(logfile,'("* : Iterations to Constrained Optimization - ",1(2x,i4))')n
103  nebrestrend = .false.
104  if ( vopttype .eq. 'lammps' .or. vopttype .eq. 'orca' .or. vopttype .eq. 'molpro' ) then
105  write(logfile,'("* : '//trim(adjustl(vopttype))//' Optimization - ")')
106  Call abinitio(mcx(m1),'optg',ldum)
107  write(logfile,'("* : '//trim(adjustl(vopttype))//' Optimization Done!")')
108  else
109  !write(logfile,'("* : Running CDE Optimization (Fire) - ")')
110  !Call FireMinimise(mcx(m1),n,gdsrestspring, nbstrength, nbrange, kradius, nebrestrend )
111  write(logfile,'("* : Running CDE Optimization (BFGS) - ")')
112  Call dfpminimise(mcx(m1),n,nebrestrend )
113  write(logfile,'("* : Iterations to CDE Optimization - ",1(2x,i4))')n
114  endif
115  nebrestrend = .true.
116  else
117  if ( vopttype .eq. 'lammps' .or. vopttype .eq. 'orca' .or. vopttype .eq. 'molpro' ) then
118  write(logfile,'("* : '//trim(adjustl(vopttype))//' Optimization - ")')
119  Call abinitio(mcx(m1),'optg',ldum)
120  write(logfile,'("* : '//trim(adjustl(vopttype))//' Optimization Done2!")')
121  elsE
122  !write(logfile,'("* : Running CDE Optimization (Fire) - ")')
123  !Call FireMinimise(mcx(m1),n,gdsrestspring, nbstrength, nbrange, kradius, nebrestrend )
124  write(logfile,'("* : Running CDE Optimization (BFGS) - ")')
125  call dfpminimise(mcx(m1),n,nebrestrend )
126  write(logfile,'("* : Iterations to CDE Optimization - ",1(2x,i4))')n
127  endif
128  !Call PrintCXSToFile( mcx(m1), 'optimized.xyz',0.0d0)
129  endif
130  ! performing checks on molecule
131  Call getgraph( mcx(m1) )
132  !print*, 'GOT GRAPHS'
133  do l = 1, naa-1
134  do j = l+1, naa
135  if (mcx(m1)%graph(l,j) /= ograph(l,j)) then
136  matchgraph = .false.
137  if ( mcx(m1)%graph(l,j) .eq. 0 ) then
138  write(logfile,'("* : Extra Bond Broken ",2X,A2,"-",A2)') mcx(m1)%atomlabel(l), mcx(m1)%atomlabel(j)
139  else
140  write(logfile,'("* : Extra Bond Formed ",2X,A2,"-",A2)') mcx(m1)%atomlabel(l), mcx(m1)%atomlabel(j)
141  endif
142  endif
143  enddo
144  enddo
145  Call getmols ( mcx(m1) )
146  !print*, 'GOT MOLS'
147  if ( mcx(m1)%nmol .gt. 1 ) write(logfile,'("* : Molecule Broke into ",I2," fragments ")') mcx(m1)%nmol
148  ! rewrites the global "reactive" array to fit the molecule in question...IMPORTANT TO COPY reactive back from backupreactive after
149  backupreactive = reactive ; reactive = .false.
150  do j = 1, cx%namol(m) ! for atoms in molecule
151  reactive(j) = backupreactive(cx%molid(m,j))
152  enddo
153  ! perform all background checks on this molecule...
154  if ( .not. moleculevalid( mcx(m1), ograph ) .or. mcx(m1)%nmol .gt. 5 ) then
155  if (mcx(m1)%nmol .gt. 5 ) write(logfile,'("* : Too many fragments, baling out ")')
156  success = .false.
157  reactive = backupreactive
158  return
159  endif
160  reactive = backupreactive
161 
162  ! if some of the atoms in this molecule are frozen, then no need to rotate
163  !
164  dontrotate = .false.
165  do l = 1, mcx(m1)%namol(1) ! atoms in molecule
166  if (mcx(m1)%fixedatom(mcx(m1)%molid(1,l))) then
167  dontrotate = .true.
168  exit
169  endif
170  enddo
171  if (.not. dontrotate) then
172  do l = 1, naa
173  cm1(1:3) = cm1(1:3) + or(1:3,l)*mcx(m1)%mass(l)
174  cm2(1:3) = cm2(1:3) + mcx(m1)%r(1:3,l)*mcx(m1)%mass(l)
175  enddo
176  cm1 = cm1/ma
177  cm2 = cm2/ma
178  do l = 1, nd/3
179  mcx(m1)%r(1:3,l) = mcx(m1)%r(1:3,l)-cm2
180  or(1:3,l) = or(1:3,l)-cm1
181  enddo
182  ! for debugging purposes..
183  ! Call GetEulerMatrices(D,pivalue/2.0d0,pivalue/3.0d0,pivalue-0.10d0)
184  ! mcx(m1)%r =(matmul(transpose(D),(mcx(m1)%r)))
185  d = matmul((or),transpose(mcx(m1)%r))
186 ! call dgesvd(jobu,jobvt, m, n, a, lda, s, u, ldu, vt, ldvt, work, lwork, info )
187  Call dgesvd( 'A', 'A', 3, 3, d(1:3,1:3), 3, e(1:3), u(1:3,1:3), 3, vt(1:3,1:3),3, work(1:100*nd), 100*nd, infos )
188  ! recycling D
189  d = matmul(u,vt)
190  if (finddet(d,3) .lt. 0.0d0 ) d = matmul((u),( matmul(unt,vt)))
191  if (infos .ne. 0) stop 'ERROR DURING SVD '
192  mcx(m1)%r = (matmul(d,mcx(m1)%r))
193  do l = 1, nd/3
194  mcx(m1)%r(1:3,l) = mcx(m1)%r(1:3,l)+cm1
195  enddo
196  endif ! dontrotate
197  enddo
198  ! put molecules back in place:
199  !
200  do m1 = 1, nopt ; m = mid(m1) ! for active molecules in fron and tail
201  j = 0
202  do l = 1, cx%namol(m) ! atoms in molecule
203  j = j + 1
204  cx%r(1:3,cx%molid(m,l)) = mcx(m1)%r(1:3,j)
205  enddo
206  enddo
207  if (.not. matchgraph) then
208  write(logfile,'("* : Optimisation did not match graph, switching imove = -1")')
209  ! Setup bondchange, atomschange
210 
211  !!for debugging purposes...
212  !write (ww,'(I3.3)') itmp
213  !filen = "testori1_"//trim(adjustl(ww))//'.xyz'
214  !call PrintCXSToFile(cx),filen,0.0d0)
215  ! get new graph... copying from mcx stops any errors.
216  do m1 = 1, nopt ; m = mid(m1) ! for active molecules in fron and tail
217  do j = 1, cx%namol(m)
218  j1 = cx%molid(m,j)
219  do l = 1, cx%namol(m)
220  l1 = cx%molid(m,l)
221  cx%graph(j1,l1) = mcx(m1)%graph(j,l)
222  enddo
223  enddo
224  enddo
225  ! get new molecules
226  call getmols ( cx )
227  endif
228  ! push the molecules that are not involved in the reaction away from the freshly optimized ones..
229  !
230  call gcminimisecollision(cx, ngdsrelax, gdsdtrelax, kradius )
231 
232  return
233  end Subroutine optimisemolecules
