386  logical :: change, esscheck

389  integer :: ix(NAMOVEMAX), iflag, rxindex(NAMAX), nrx, ii

391  change = .false.
392 
393  ! Decide which end-point to change.
394  !
395  if (ran2(irun,0.d0,1.d0) < 0.5d0) then

421  !

447 99 ii = int( dble(nrx) * ran2(irun,0.d0,1.d0)) + 1

464  enddo
465  enddo

467 
468  ! Check that the selected atoms match the graph.

485  else

487  endif
488  endif
489  enddo

494  esscheck = .false.

503  endif
