357  Subroutine graphmoves( rp, change, ic, imove )
358  implicit none
359  type(cxs) :: cx, cxstore
360  type(rxp) :: rp
361  integer :: ic, i, imove, itry, j, k, itry2, sum
362  logical, allocatable :: bondchange(:,:), atomchange(:)
363  logical :: change, esscheck
364  real(8) :: ran2, rx
365  character (len=2) :: id1, id2, i1, i2
366  integer :: ix(NAMOVEMAX), iflag, rxindex(NAMAX), nrx, ii
367 
368  change = .false.
369 
370  ! Decide which end-point to change.
371  !
372  if (ran2(irun,0.d0,1.d0) < 0.5d0) then
373  ic = 1
374  else
375  ic = rp%nimage
376  endif
377 
378 
379  ! Create a copy of the chemical structure being changed.
380  !
381  Call copytonewcxs( rp%cx(ic), cxstore )
382 
383 
384  ! Set number of atoms.
385  !
386  na = rp%na
387 
388 
389  ! Setup bondchange and atomchange matrices.
390  !
391  allocate( bondchange(na,na) )
392  allocate( atomchange(na) )
393 
394  atomchange(:) = .false.
395  bondchange(:,:) = .true.
396 
397 
398  ! First, use the information from the input file to define which atoms
399  ! can react, and which bonds can change.
400  !
401  ! First, check reactive atom types...
402  !
403 
404  do i = 1, na
405  do j = 1, nreactivetypes
406  if ( reactivetype(j) == cxstore%atomlabel(i) ) then
407  atomchange(i) = .true.
408  endif
409  enddo
410  enddo
411 
412 
413  ! Second, check reactive atom ranges and ids.
414  !
415  do i = 1, na
416  atomchange(i) = reactive(i)
417  enddo
418 
419 
420  ! Third, fix indicated bonds.
421  !
422  do i = 1, na
423  do j = 1, na
424  if (.not.fixedbonds(i,j)) then
425  bondchange(i,j) = .true.
426  endif
427  enddo
428  enddo
429  do i = 1, na
430  id1 = cxstore%atomlabel(i)
431  do j = 1, na
432  id2 = cxstore%atomlabel(j)
433  do k = 1, nfixtype
434  i1 = fixedbondtype(k,1)
435  i2 = fixedbondtype(k,2)
436 
437  if (trim(id1) == trim(i1) .and. trim(id2) == trim(i2) ) then
438  bondchange(i,j) = .false.
439  else if (trim(id1) == trim(i2) .and. trim(id2) == trim(i1) ) then
440  bondchange(i,j) = .false.
441  endif
442  enddo
443  enddo
444  bondchange(i,i) = .false.
445  enddo
446 
447 
448  ! Finally, check the valence constraints of the atoms..
449  !
450  do i = 1, na
451 
452  do k = 1, nrxval
453  if (cxstore%atomlabel(i) == trim(rxvalatom(k,1))) then
454 
455  ! Sum up bonded atoms.
456  !
457  sum = 0
458  do j = 1, na
459  if (cxstore%atomlabel(j) == trim(rxvalatom(k,2))) then
460  if (i/=j) then
461  sum = sum + cxstore%graph(i,j)
462  endif
463  endif
464  enddo
465  if (sum < rxvalrange(k,1) .or. sum > rxvalrange(k,2))then
466  atomchange(i) = .false.
467  endif
468  endif
469  enddo
470  enddo
471 
472 
473  ! Set up an array of reactive atoms to help with random selection.
474  !
475  nrx = 0
476  do i = 1, na
477  if (atomchange(i)) then
478  nrx = nrx + 1
479  rxindex(nrx) = i
480  endif
481  enddo
482 
483 
484  ! Now, keep trying moves until we hit NTRYMAX....
485  !
486  itry = 0
487  iflag = 0
488  outer: do while ( itry < ntrymax )
489 
490  itry = itry + 1
491 
492  ! Select a move type based on probabilities.
493  !
494  rx = ran2(irun,0.d0,1.d0)
495  imove = int( dble(ngmove) * ran2(irun,0.d0,1.d0) ) + 1
496 
497  if ( rx > moveprob(imove) ) cycle outer
498 
499 
500  ! Now try to find namove(imove) atoms which match the target start graph.
501  !
502  itry2 = 0
503  inner: do while (itry2 < ntrymax2)
504 
505  itry2 = itry2 + 1
506 
507 
508  ! Select atoms from atomchange() list.
509  !
510  do i = 1, namove(imove)
511 99 ii = int( dble(nrx) * ran2(irun,0.d0,1.d0)) + 1
512  ix(i) = rxindex(ii)
513  do j = 1, i-1
514  if (ix(i) == ix(j))goto 99
515  enddo
516  enddo
517 
518 
519  ! Check reactivity flags
520  !
521  do i = 1, namove(imove)
522  if (.not.atomchange(ix(i)))cycle inner
523  enddo
524 
525  do i = 1, namove(imove)-1
526  do j = i+1, namove(imove)
527  if (.not.bondchange(ix(i),ix(j))) cycle inner
528  enddo
529  enddo
530 
531 
532  ! Check that the selected atoms match the graph.
533  !
534  do i = 1, namove(imove)
535  do j = 1, namove(imove)
536  if (cxstore%graph(ix(i),ix(j)) /= gmstart(imove,i,j)) cycle inner
537  enddo
538  enddo
539 
540 
541  ! Check that the labels match the target labels....
542  !
543  !movelabel(ngmove,i),i=1,namove(ngmove)
544  !
545  do i = 1, namove(imove)
546  if (trim( movelabel(imove,i) ) /= '*') then
547  if (trim(movelabel(imove,i)(1:2)) == '*!') then
548  if (trim(movelabel(imove,i)(3:)) == trim(cxstore%atomlabel(ix(i)))) cycle inner
549  else
550  if (trim(movelabel(imove,i)) /= trim(cxstore%atomlabel(ix(i))) ) cycle inner
551  endif
552  endif
553  enddo
554 
555  ! Check that the moves involve the essential atoms
556  !
557  do i = 1, cxstore%na
558  if ( essential(i) ) then
559  esscheck = .true.
560  do j = 1, namove(imove)! no of atom in move
561  do k = 1, cxstore%nmol! no of molecules
562  if ( any(cxstore%molid(k,:) == ix(j)) .and. any(cxstore%molid(k,:) == i) ) then
563  esscheck = .false.
564  endif
565  enddo
566  enddo
567  if ( esscheck ) cycle inner
568  endif
569  enddo
570 
571 
572 
573  ! Perform the graph move by updating the graph.
574  !
575  do i = 1, namove(imove)
576  do j = 1, namove(imove)
577 
578  cxstore%graph(ix(i), ix(j)) = gmend(imove,i,j)
579  enddo
580  enddo
581  iflag = 1
582  exit inner
583 
584  enddo inner
585 
586 
587  if (iflag == 1) then
588 
589 
590  ! Check that the final valences are sensible.
591  !
592  do i = 1, na
593  sum = 0
594  do j = 1, na
595  if (i/=j) then
596  sum = sum + cxstore%graph(i,j)
597  endif
598  enddo
599 
600  do k = 1, nvalcon
601  if (trim(cxstore%atomlabel(i)) == trim(valatom(k))) then
602  if (sum < valrange(k,1) .or. sum > valrange(k,2))then
603  Call copycxs(rp%cx(ic),cxstore)
604  iflag = 0
605  cycle outer
606  endif
607  endif
608  enddo
609  enddo
610 
611 
612  ! Check that any allowed bonding constraints are not violated.
613  !
614  do i = 1, na
615  do k = 1, nallowbonds
616  if (trim(cxstore%atomlabel(i)) == trim( allowbondsatom(k,1) )) then
617  sum = 0
618  do j = 1, na
619  if (i/=j .and. (trim(cxstore%atomlabel(j)) == trim(allowbondsatom(k,2))) ) then
620  sum = sum + cxstore%graph(i,j)
621  endif
622  enddo
623  if (sum > allowbondsmax(k)) then
624  Call copycxs(rp%cx(ic),cxstore)
625  iflag = 0
626  cycle outer
627  endif
628  endif
629  enddo
630  enddo
631 
632 
633  ! If we get here, everything is OK!
634  !
635  write(logfile,'("* Graph move accepted: ",1x,i4)')imove
636  write(logfile,'("* Updated atoms:")')
637  do i = 1, namove(imove)
638  write(logfile,*)ix(i),cxstore%atomlabel(ix(i))
639  enddo
640  change = .true.
641 
642  exit outer
643 
644  endif
645 
646  enddo outer
647 
648 
649  ! If we've changed the graph, update the path.
650  !
651  if (change) then
652  Call copycxs(cxstore, rp%cx(ic))
653  endif
654 
655 
656  ! Clean up memory.
657  !
658  deallocate( atomchange, bondchange )
659 
660 
661  return
662  end Subroutine graphmoves
