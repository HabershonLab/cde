262  ! matrix of molecules with atoms bonding/ breaking during a reaction
263  ! if k1 = k2, id gives the id of atoms that molecule m1 shares to (if aid(id,1)) or takes from (if aid(id,2)) molecule m2 in the same image
264  ! if k1 /= k2, id gives similar information but for inter-images
265  ! ic gives information about what atoms of molecule m1 in image k are shared (are common, overlap) with molecule m2 of complementary image
266  type molgm
267  integer, allocatable :: id(:,:,:,:) ! the ID for naid and aid, given indecies id(k1,k2,m1,m2), where m1 = molecule id of image k1, m2 = molecule id of image k2
268  integer, allocatable :: naid(:,:) ! the number of atoms that these two molecules exchange naid(id,1:2)
269  integer, allocatable :: aid(:,:,:) ! the id of the atoms these two molecules exchange, now the last index aid(id,naid,1:2) =1 corresponds to the atom in molecule m1, =2 atom in molecule m2

271  integer :: na ! fourth dimension of aid and dir
